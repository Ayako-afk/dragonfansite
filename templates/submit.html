@app.route("/submit", methods=["GET", "POST"]) def submit(): if request.method == "POST": # Get the username entered by the user username = request.form.get("username") # If no username is provided, return an error message if not username: return "Error: Please enter a username" # Construct the URL to fetch the user's growing dragons url = f"https://dragcave.net/api/v2/user?username={username}&filter=GROWING" # Add the User-Agent header (important for Dragon Cave to allow requests) headers = {"User-Agent": "Mozilla/5.0"} try: # Make the API request to fetch dragons resp = requests.get(url, headers=headers, timeout=10) except requests.RequestException as e: # Handle any network or request errors return f"Error fetching user: {e}" # Handle 401 Unauthorized (private account or bad request) if resp.status_code == 401: return f"Error: User '{username}' is private or unauthorized" # Handle other HTTP errors (non-200 status codes) elif resp.status_code != 200: return f"Error fetching user: {resp.status_code}" # Parse the JSON response data = resp.json() # Check if the user has dragons and handle if no dragons are found if "dragons" not in data or len(data["dragons"]) == 0: return f"No growing dragons found for user '{username}'" # Load existing dragons from the JSON file dragons = load_dragons() # Append each dragon to the list, avoiding duplicates for dragon in data["dragons"]: if not any(d["id"] == dragon["id"] for d in dragons): # Avoid duplicates dragons.append({ "owner": username, "id": dragon["id"], "name": dragon.get("name") or "Unnamed", "image": dragon.get("image", "") }) # Save the updated list of dragons to the file save_dragons(dragons) # Redirect back to the index page to show the updated dragon list return redirect(url_for("index")) # If it's a GET request, render the form to add a new user return render_template("submit.html")
